#| echo: false
#| label: fig-emaranhamento-bulk
#| fig-cap: "Representação esquemática do *bulk* (AdS) com contornos vazados, dividindo as regiões $A$ e $B$ com a superfície homóloga mínima $\\tilde{A}$."
#| fig-align: center

import numpy as np
import matplotlib.pyplot as plt

fig = plt.figure(figsize=(7, 7))
ax = fig.add_subplot(111, projection='3d')

# 1. Wireframe da Esfera Externa (Vazado)
u = np.linspace(0, 2 * np.pi, 100)
v = np.linspace(0, np.pi, 100)
x = np.outer(np.cos(u), np.sin(v))
y = np.outer(np.sin(u), np.sin(v))
z = np.outer(np.ones(np.size(u)), np.cos(v))

# Plota a casca da esfera (fio de arame super leve para dar volume)
ax.plot_surface(x, y, z, color='whitesmoke', alpha=0.4, edgecolor='lightgray', linewidth=0.3)

# 2. Contornos Principais em Preto
theta = np.linspace(0, 2 * np.pi, 150)

# Equador (Plano XY) dividindo A e B na fronteira
ax.plot(np.cos(theta), np.sin(theta), np.zeros_like(theta), color='black', linewidth=1.5)

# Contorno da Superfície Divisória (Fronteira de A-til no Plano YZ)
ax.plot(np.zeros_like(theta), np.cos(theta), np.sin(theta), color='black', linewidth=1.5)

# 3. Hachuras da Superfície Mínima (presas estritamente dentro do corte)
amplitude = 0.25 
for z_line in np.linspace(-0.85, 0.85, 11):
    y_max = np.sqrt(1 - z_line**2)
    y_line = np.linspace(-y_max, y_max, 50)
    x_curve = amplitude * (1 - (y_line**2 + z_line**2))
    ax.plot(x_curve, y_line, z_line * np.ones_like(y_line), color='black', linewidth=1)

# 4. Textos Centralizados e Alinhados
ax.text(-1.3, .4, 0, 'B', color='black', fontsize=24, fontfamily='serif', ha='center', va='center')
ax.text(1.3, 0, 0, 'A', color='black', fontsize=24, fontfamily='serif', ha='center', va='center')

# A-til movido para o centro geométrico da membrana hachurada
ax.text(0.3, 0, 0, r'$\tilde{A}$', color='black', fontsize=26, fontfamily='serif', ha='center', va='center')

# 5. Ajustes de Câmera e Ocultação de Eixos
ax.view_init(elev=20, azim=35) 
ax.set_aspect('equal')
ax.axis('off')

plt.tight_layout()
plt.show()
